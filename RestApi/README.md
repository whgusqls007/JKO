<h3>Install Django</h3>
    <pre>
        pip install django
    </pre>
<h3>Install Djongo</h3>
    <pre>
        pip install djongo
    </pre>
<h3>Install DjangoRestFramework</h3>
    <pre>
        pip install djangorestframework
    </pre>
<h3>Start server</h3>
    <pre>
        python manage.py runserver IP:PORT
    </pre>
